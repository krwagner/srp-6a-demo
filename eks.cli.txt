#
# Deploy ECS via CLI
#

# Create New EKS Cluster:
aws eks create-cluster --name eksworkshop-cf --role-arn "arn:aws:iam::777191436106:role/eksServiceRole" --resources-vpc-config subnetIds="subnet-05e23d0a1f8cf2416,subnet-03e960a23c2a6e6b1",securityGroupIds="sg-0b0c0d1d569fbffaf"

# Get Certificate Authority Data:
aws eks describe-cluster --name eksworkshop-cf  --query cluster.[certificateAuthority.data] --output text

# Get Cluster Endpoint:
aws eks describe-cluster --name eksworkshop-cf  --query cluster.[endpoint] --output=text
